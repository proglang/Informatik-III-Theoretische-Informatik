\section[Rekursive Funktionen]{Rekursive Funktionen}
\label{sec:rekursive-funktionen}

\newcommand{\sub}{\textrm{sub}}
\newcommand{\pred}{\textrm{pred}}

Turingmaschinen beschreiben Berechnungen durch einzelne
Konfigurationsübergänge. In diesem Kapitel nehmen wir eine funktionale
Perspektive ein: Aus wenigen einfachen Funktionen über $\N$ erzeugen wir durch
fest vorgegebene Konstruktionsprinzipien immer größere Klassen berechenbarer
Funktionen.

Zunächst führt primitive Rekursion ausschließlich zu totalen Funktionen. Durch
Minimierung kommt partielle Berechenbarkeit hinzu, und wir erhalten genau die
mit Turingmaschinen berechenbaren Funktionen. Die Programmiersprachen LOOP und
WHILE stellen anschließend eine Verbindung zwischen diesen abstrakten
Funktionsklassen und konkreten Berechnungsabläufen her. Zum Abschluss zeigt die
Ackermannfunktion, dass nicht jede totale berechenbare Funktion bereits primitiv
rekursiv ist.

\emph{Notation:} Mit $f: A \to B$ bezeichnen wir eine \emph{totale}
Funktion von $A$ nach $B$ und mit $f:A\hookrightarrow B$ eine
\emph{partielle} Funktion von $A$ nach $B$. Jede totale Funktion kann
auch als partielle Funktion betrachtet werden.

Als erstes Konstruktionsprinzip betrachten wir die primitive Rekursion. Dabei
wird das erste Argument schrittweise aufgebaut, während die übrigen Argumente
als Parameter unverändert bleiben.

\begin{Def}[Schema der primitiven Rekursion]
	\begin{align*}
		\text{Sei } g&: \N^k \-> \N \qquad,\ k\geq 0\\
		h&: \N^{k+2}\->\N\\
		\text{Falls eine Funktion } f&: \N^{k+1}\->\N\text{ die folgenden Gleichungen erfüllt:}\\
		f(0,\overline{x}) &= g(\overline{x}) \qquad,\ \overline{x}\in\N^k\\
		\begin{split}
			f(n+1,\overline{x}) &= h(f(n,\overline{x}),n,\overline{x})
		\end{split} \numbereq\label{eq:rekursives f}
	\end{align*}
	Dann sei
	\[ f =: PR(g,h) \]
	aus $g$ und $h$ durch primitive Rekursion definiert.
\end{Def}
Die Rekursionsgleichungen sollen nicht nur eine Berechnungsvorschrift andeuten,
sondern tatsächlich genau eine totale Funktion festlegen. Dies ist Inhalt des
folgenden Lemmas.
\begin{lemma}[name={[$PR(g,h)$ ist wohldefiniert.]}]
  \label{lemma:rekursive-funktionen-1}
  $PR(g,h)$ ist wohldefiniert.
\end{lemma}
\begin{proof}
	Seien $g:\N^k\to\N$ und $h:\N^{k+2}\to\N$ wie in der Definition.
	Wir definieren $F\subseteq(\N\times\N^k)\times\N$ als die von den
	folgenden Regeln induktiv erzeugte Relation:
	\[
		\frac{\strut}{((0,\overline{x}),g(\overline{x}))\in F}
		\qquad
		\frac{((n,\overline{x}),y)\in F}
		     {((n+1,\overline{x}),h(y,n,\overline{x}))\in F}.
	\]
	Dabei besteht $F$ genau aus den Tupeln, die sich durch endlich viele
	Anwendungen dieser Regeln herleiten lassen. Insbesondere gelten die
	folgenden Inversionseigenschaften:
	\begin{align*}
		((0,\overline{x}),y)\in F
		&\quad\Longrightarrow\quad y=g(\overline{x}),\\
		((n+1,\overline{x}),y)\in F
		&\quad\Longrightarrow\quad
		\exists z\in\N:\
		((n,\overline{x}),z)\in F
		\ \land\ y=h(z,n,\overline{x}).
	\end{align*}

	\begin{description}
	\item[Totalität]
		Wir zeigen durch Induktion über $n$, dass für alle
		$\overline{x}\in\N^k$ ein $y\in\N$ mit
		$((n,\overline{x}),y)\in F$ existiert.
		Für $n=0$ liefert die erste Regel das Element
		$((0,\overline{x}),g(\overline{x}))$.
		Gibt es für $n$ ein $y$ mit $((n,\overline{x}),y)\in F$, so liefert
		die zweite Regel
		$((n+1,\overline{x}),h(y,n,\overline{x}))\in F$.

	\item[Funktionalität]
		Wir zeigen ebenfalls durch Induktion über $n$, dass aus
		\[
			((n,\overline{x}),y)\in F
			\quad\text{und}\quad
			((n,\overline{x}),y')\in F
		\]
		stets $y=y'$ folgt.
		Für $n=0$ ergeben die Inversionseigenschaften
		$y=g(\overline{x})=y'$.
		Für den Induktionsschritt gibt es $z,z'\in\N$ mit
		\begin{align*}
			((n,\overline{x}),z)&\in F,
			&y&=h(z,n,\overline{x}),\\
			((n,\overline{x}),z')&\in F,
			&y'&=h(z',n,\overline{x}).
		\end{align*}
		Nach Induktionsvoraussetzung ist $z=z'$ und damit auch $y=y'$.
	\end{description}

	Da $F$ total und funktional ist, definiert
	\[
		f(n,\overline{x})=y
		\quad\Longleftrightarrow\quad
		((n,\overline{x}),y)\in F
	\]
	eine totale Funktion $f:\N^{k+1}\to\N$.

	\begin{description}
	\item[Rekursionsgleichungen]
		Nach der ersten Erzeugungsregel gilt
		$((0,\overline{x}),g(\overline{x}))\in F$ und somit
		$f(0,\overline{x})=g(\overline{x})$.
		Aus $((n,\overline{x}),f(n,\overline{x}))\in F$ folgt nach der zweiten
		Regel
		\[
			((n+1,\overline{x}),
			h(f(n,\overline{x}),n,\overline{x}))\in F.
		\]
		Aufgrund der Funktionalität von $F$ gilt daher
		\[
			f(n+1,\overline{x})
			=h(f(n,\overline{x}),n,\overline{x}).
		\]
		Damit erfüllt $f$ beide Gleichungen aus \eqref{eq:rekursives f}.

	\item[Eindeutigkeit]
		Sei $f':\N^{k+1}\to\N$ eine weitere Funktion, die
		\eqref{eq:rekursives f} erfüllt. Ihr Graph
		\[
			G_{f'}
			:=\{((n,\overline{x}),f'(n,\overline{x}))
			\mid n\in\N,\ \overline{x}\in\N^k\}
		\]
		ist unter beiden Erzeugungsregeln abgeschlossen: Dies folgt für die
		erste Regel aus der Anfangsgleichung und für die zweite Regel aus der
		Rekursionsgleichung von $f'$.
		Da $F$ die kleinste unter diesen Regeln abgeschlossene Relation ist,
		gilt $F\subseteq G_{f'}$. Folglich ist für alle $n$ und
		$\overline{x}$ auch $f(n,\overline{x})=f'(n,\overline{x})$.
	\end{description}
\end{proof}
Mit dem wohldefinierten Rekursionsschema können wir nun die erste
Funktionsklasse durch Basisfunktionen und Abschlussoperationen festlegen.
\begin{Def}[name={[\acs*{PREC}]}]\label{def:PREC}
	Die \acf{PREC} über $\N$ ist induktiv definiert:
	\begin{enumerate}
	\item\label{itm:Nullfunktion} $\forall k\in\N:
		\begin{aligned}[t]
			&\mathcal{O}^k : \N^k\->\N\\
			&\mathcal{O}^k(\overline{x}) = 0\\
		\end{aligned}$\\
		$\mathcal{O}^k\in\ac{PREC}$ (die Nullfunktion)
	\item\label{itm:Projektion} $\forall k\geq1: \forall 1\leq j\leq k:
		\begin{aligned}[t]
			&\pi^k_j : \N^k\->\N\\
			\pi^k_j(x_1,\dots,x_k) = x_j
		\end{aligned}$\\
		$\pi^k_j\in\ac{PREC}$ (Projektionen)
	\item\label{itm:Nachfolgerfunktion} $S:\N\->\N\ \in\ac{PREC}$ (Nachfolgerfunktion)
	\pagebreak[3]
	\item\label{itm:Komposition} Falls $g:\N^k\->\N\ \in\ac{PREC}$
		\begin{align*}
			\forall 1\leq i\leq k: h_i:\N^m\->\N \quad&\in\ac{PREC}\\
			\text{dann ist }g\circ[h_1,\dots,h_k]:\N^m\->\N \quad&\in\ac{PREC}\\
			(g\circ[h_1,\dots,h_k])(\overline{x}) = g(h_1(\overline{x}),\dots,h_k(\overline{x})) \quad\x&\in\N^m
		\end{align*}
		(Funktionskomposition)
	\item\label{itm:Rekursion} Falls $g:\N^k\->\N$ und $h:\N^{k+2}\->\N\quad\in\ac{PREC}$\\
		dann ist auch $PR(g,h)\in\ac{PREC}$ \quad(Rekursion) \qedhere
	\end{enumerate}
\end{Def}
Die folgenden Konstruktionen zeigen, wie sich vertraute arithmetische
Operationen allein aus diesen Bausteinen zusammensetzen lassen.
\begin{Bsp*}\
  \begin{enumerate}
  \item Addition $\in\ac{PREC}$
    \begin{align*}
      \add(0,x) &=x\\
      \add(n+1,x) &= \add(n,x)+1\\
                &= h(\add(n,x),n,x)\\
      \add &= PR(g,h)\\
      \text{mit }g(x) &= x \quad ; g=\pi^1_1\\
      h(y,n,x) &= y+1 \quad ; h= S\circ \pi^3_1\\
      \add &= PR(\pi^1_1,S\circ\pi^3_1)
    \end{align*}
  \item Multiplikation
    \begin{alignat*}{2}
      &&\mult&: \N^2\->\N\\
      &&\mult(0,x) &=0\\
      &&\mult(n+1,x) &= \add(x,\mult(n,x))\\
      &&\mult &= PR(g,h)\\
      \text{mit}&\ & g&= \mathcal{O}^1\\
      &&h(y,n,x) &= \add(x,y)\\
      \text{d.h.}&& h &=\add\circ[\pi^3_3,\pi^3_1]
    \end{alignat*}
  \item Subtraktion auf natürlichen Zahlen ist
    nicht total. Stattdessen verwenden wir die \emph{Monus} Operation
    definiert durch $x \dotminus y = \max (x-y, 0)$ hier definiert als
    \sub. Vorher definieren wir die Vorgängerfunktion \pred.
    \begin{align*}
      \pred&: \N \to \N &
      \pred&= PR (\mathcal{O}^0, \pi^2_2) \\
      \sub&: \N^2 \to \N &
      \sub&= PR (\pi^1_1, \pred \circ \pi^3_1) \circ [\pi^2_2, \pi^2_1]
    \end{align*}
  \end{enumerate}
\end{Bsp*}
Um mehrere natürliche Zahlen gemeinsam durch eine einzige natürliche Zahl zu
codieren, benötigen wir Paarfunktionen.
\begin{Def}
  Eine \emph{Paarfunktion} ist eine bijektive Funktion $p : \N^2 \to \N$.
\end{Def}
Die Cantorsche Paarfunktion liefert eine konkrete primitiv rekursive
Paarfunktion; auch ihre beiden Umkehrprojektionen lassen sich primitiv rekursiv
berechnen.
\begin{Bsp}
  Die Cantorsche Paarfunktion $cp$ ist wie folgt definiert.
  \begin{align*}
    tr & : \N \to \N  & tr (0) &= 0 & tr(n+1) &= n+1 + tr(n) \\
    cp & : \N^2 \to \N & cp (x, y) &= tr (x+y) + y
  \end{align*}
  Offenbar ist diese Funktion primitiv rekursiv:
  \begin{align*}
    tr &= PR (\mathcal{O}^0, \add \circ [S\circ \pi^2_2, \pi^2_1])\\
    cp &= \add \circ [tr\circ[\add], \pi^2_2]
  \end{align*}
  Die diagonale Aufzählung von $\N^2$ durch $cp$ ist in
  \autoref{fig:cantor-pairing} dargestellt.

  \begin{figure}[tp]

  \end{figure}

  Die beiden Projektionen $px, py : \N \to \N$ mit den Eigenschaften
  (für alle $x, y, n \in \N$)
  \begin{align*}
    x &= px (cp (x, y)) & y &= py (cp (x, y)) & n & = cp (px (n), py (n))
  \end{align*}
  sind ebenfalls primitiv rekursiv.

  Dazu definiere eine primitiv rekursive Funktion $untr : \N \to \N$,
  sodass $m = untr(n)$ impliziert $tr(m)\le n$ und $n <
  tr(m+1)$. (Übung oder selbst)

  Dann setze
  \begin{align*}
    py (n) &= n \dotminus tr (untr (n)) &
    px (n) &= untr (n) \dotminus py (n)
  \end{align*}

\end{Bsp}
Alle bisherigen Konstruktionen terminieren für jede Eingabe. Diese Totalität
ist eine grundlegende, aber zugleich einschränkende Eigenschaft der primitiven
Rekursion.
\begin{Beobachtung}
	Alle primitiv rekursiven Funktionen sind total.\\
	Aber: $\exists$ totale Funktionen, die nicht primitiv rekursiv sind.
\end{Beobachtung}
Um auch nicht terminierende Berechnungen zu beschreiben, führen wir nun eine
potenziell partielle Suchoperation ein. Sie sucht die erste Nullstelle, darf
dabei aber keinen undefinierten früheren Funktionswert überspringen.
\begin{Def}[Minimierung]\label{def:Minimierung}
	Sei $f:\N^{k+1}\hookrightarrow\N$ partiell. Für
	$\overline{x}\in\N^k$ sei
	\[
		C_f(\overline{x})
		:=\{n\in\N \mid f(n,\overline{x})=0\text{ und }
		\forall j<n:\ f(j,\overline{x})\downarrow\ne0\}.
	\]
	Dabei bedeutet $f(j,\overline{x})\downarrow$, dass
	$f(j,\overline{x})$ definiert ist. Die \emph{Minimierung} von $f$ ist
	durch die folgende Vorschrift gegeben:
	\[
		(\mu f)(\overline{x})
		=
		\begin{cases}
			\min C_f(\overline{x}), & C_f(\overline{x})\ne\emptyset,\\
			\text{undefiniert}, & C_f(\overline{x})=\emptyset.
		\end{cases}
	\]
\end{Def}

Da die Ausgangsfunktion partiell sein darf, ist zu prüfen, ob diese Vorschrift
wirklich eine eindeutig bestimmte partielle Funktion liefert.
\begin{Satz}[name={Wohldefiniertheit der Minimierung}]
	\label{satz:Minimierung-wohldefiniert}
	Die Vorschrift aus \autoref{def:Minimierung} bestimmt zu jeder partiellen
	Funktion $f:\N^{k+1}\hookrightarrow\N$ eindeutig eine partielle Funktion
	$\mu f:\N^k\hookrightarrow\N$.
\end{Satz}
\begin{proof}
	Sei $\overline{x}\in\N^k$ fest. Falls
	$C_f(\overline{x})=\emptyset$ ist, weist die Vorschrift
	$\overline{x}$ keinen Funktionswert zu. Andernfalls besitzt
	$C_f(\overline{x})$ nach dem Wohlordnungsprinzip für $\N$ ein eindeutig
	bestimmtes kleinstes Element.

	Tatsächlich kann $C_f(\overline{x})$ sogar höchstens ein Element enthalten:
	Wären $n,m\in C_f(\overline{x})$ mit $n<m$, so wäre einerseits
	$f(n,\overline{x})=0$. Andererseits würde $m\in C_f(\overline{x})$ wegen
	$n<m$ gerade $f(n,\overline{x})\downarrow\ne0$ verlangen, ein Widerspruch.

	Somit wird jedem $\overline{x}$ entweder genau ein Wert aus $\N$ oder kein
	Wert zugewiesen. Die Vorschrift definiert also eine partielle Funktion.
	Sowohl ihr Definitionsbereich
	$\{\overline{x}\mid C_f(\overline{x})\ne\emptyset\}$ als auch ihr Wert auf
	diesem Definitionsbereich sind durch $f$ festgelegt. Daher ist diese
	partielle Funktion eindeutig bestimmt.
\end{proof}

Aus dem Beweis ergeben sich unmittelbar die folgenden nützlichen Fälle für
definierte und undefinierte Minimierungen.
\begin{Beobachtung}
	Für $\overline{x}\in\N^k$ folgen unmittelbar aus der Definition:
	\begin{enumerate}
	\item Gibt es kein $n\in\N$, für das $f(n,\overline{x})$ definiert und
		gleich $0$ ist, so ist $(\mu f)(\overline{x})$ undefiniert.
	\item Sei $n_0$ das kleinste $n$, für das $f(n,\overline{x})$ definiert und
		gleich $0$ ist. Gibt es ein $j<n_0$, für das $f(j,\overline{x})$
		undefiniert ist, so ist $(\mu f)(\overline{x})$ undefiniert.
	\item Gilt $f(n,\overline{x})=0$ und
		$f(j,\overline{x})\downarrow\ne0$ für alle $j<n$, so gilt
		$(\mu f)(\overline{x})=n$.
	\end{enumerate}
\end{Beobachtung}

Für die folgende Klassendefinition verwenden wir auch Komposition und primitive
Rekursion auf partiellen Funktionen.
\begin{Bemerkung}
  Die Konstruktionen aus \autoref{def:PREC} können genauso auch für
  partielle Funktionen definiert und verwendet werden. Dabei nimmt man
  an, dass ein Funktionsaufruf nur dann definiert ist, falls auch sein
  Argument eine definierten Wert hat. Ebenso kann
  \autoref{lemma:rekursive-funktionen-1} für partielle Funktionen bewiesen werden.
\end{Bemerkung}

Ergänzen wir die primitiven rekursiven Konstruktionen um Minimierung, so
erhalten wir die Klasse der $\mu$-rekursiven Funktionen.
\begin{Def}[name={[\acs*{muREC}]}]
  Die \acf{muREC} über $\N$ ist die
  kleinste Menge von Funktionen, die die Basisfunktionen (1,2,3 aus
  der \autoref{def:PREC}) enthält und abgeschlossen ist unter
  \hyperref[itm:Komposition]{Komposition \ref*{itm:Komposition}},
  \hyperref[itm:Rekursion]{Rekursion \ref*{itm:Rekursion}} und
  \hyperref[def:Minimierung]{Minimierung}.
\end{Def}
Diese algebraisch definierte Klasse stimmt mit dem zuvor durch Maschinen
gewonnenen Berechenbarkeitsbegriff überein.
\begin{Satz}[name={[\acs*{muREC} $\hat=$ Klasse der \acs*{TM}-berechenbaren Fkt.]}]
	\ac{muREC} stimmt mit der Klasse der \ac{TM}-berechenbaren Funktionen über $\N$ überein.
\end{Satz}
\begin{proof}\
	\begin{itemize}
	\item[$\supseteq$] Sei $M$ eine deterministische \ac{TM}, die eine partielle
		Funktion $f:\N^k\hookrightarrow\N$ berechnet. Wir skizzieren, wie die
		Berechnung von $M$ durch eine Funktion aus \ac{muREC} simuliert wird.
		\begin{enumerate}
		\item Zustände, Bandsymbole, endliche Wörter und damit Konfigurationen
			$(v,q,w)$ von $M$ werden durch natürliche Zahlen codiert. Die Codierung
			kann so gewählt werden, dass die benötigten Codierungs- und
			Decodierungsfunktionen primitiv rekursiv sind.
		\item Da $M$ fest ist, sind die Funktionen
			\[
			\mathsf{start}_{M}:\N^k\->\N
			\qquad\text{und}\qquad
			\mathsf{step}_{M}:\N\->\N
			\]
			primitiv rekursiv. Sie liefern den Code der Startkonfiguration bzw. den
			Code der eindeutigen Folgekonfiguration. Für Haltekonfigurationen setzen
			wir $\mathsf{step}_{M}(c)=c$.
		\item Durch primitive Rekursion ist auch
			\begin{align*}
			\mathsf{conf}_{M}(0,\overline{x})
				&=\mathsf{start}_{M}(\overline{x}),\\
			\mathsf{conf}_{M}(n+1,\overline{x})
				&=\mathsf{step}_{M}(\mathsf{conf}_{M}(n,\overline{x}))
			\end{align*}
			primitiv rekursiv. $\mathsf{conf}_{M}(n,\overline{x})$ codiert die
			Konfiguration nach $n$ Schritten.
		\item Die charakteristische Funktion $\mathsf{halt}_{M}$ der Codes von
			Haltekonfigurationen ist primitiv rekursiv; dabei gelte
			$\mathsf{halt}_{M}(c)=0$ genau für Haltekonfigurationen. Damit ist auch
			\[
			H_{M}(n,\overline{x})
			:=\mathsf{halt}_{M}(\mathsf{conf}_{M}(n,\overline{x}))
			\]
			primitiv rekursiv. Durch Minimierung erhalten wir
			\[
			t_{M}(\overline{x})
			:=(\mu H_{M})(\overline{x}).
			\]
			$\mu$-rekursiv. Diese Funktion ist genau dann undefiniert, wenn $M$ auf
			$\overline{x}$ nicht hält.
		\item Schließlich sei $\mathsf{out}_{M}$ die primitiv rekursive Funktion,
			die aus dem Code einer Haltekonfiguration die codierte Ausgabe liest. Dann
			gilt
			\[
			f(\overline{x})=
			\mathsf{out}_{M}\bigl(
			\mathsf{conf}_{M}(t_{M}(\overline{x}),\overline{x})
			\bigr).
			\]
			Die rechte Seite entsteht durch Minimierung und Komposition aus primitiv
			rekursiven Funktionen; folglich ist $f\in\ac{muREC}$.
		\end{enumerate}
	\item[$\subseteq$] simuliere $\mu$-Rekursion mit einer \ac{TM}
		(analog zur universellen \ac{TM}). \qedhere
	\end{itemize}
\end{proof}
Die bisherigen Konstruktionsprinzipien beschreiben Funktionen auf abstrakte
Weise. LOOP-Programme geben ihnen ein einfaches imperatives Berechnungsmodell
mit ausschließlich beschränkten Schleifen gegenüber.
\begin{Def}[name={LOOP-Programme}]\label{def:LOOP-Programme}
  Ein LOOP-Programm arbeitet mit Registern
  $x_1,x_2,\dots$, deren Werte natürliche Zahlen sind. Die Menge der
  LOOP-Programme ist induktiv durch die folgende Syntax definiert:
  \[
    \begin{array}{rcll}
      P,Q &::=& x_i := x_i+1
      & \text{(Inkrementierung)},\\
          &\mid& x_i := x_i\dotminus 1
      & \text{(Dekrementierung mit Abschneiden bei $0$)},\\
          &\mid& P;Q
      & \text{(sequenzielle Ausführung)},\\
          &\mid& \texttt{LOOP }x_i\texttt{ DO }P\texttt{ END}
      & \text{(beschränkte Schleife)}.
    \end{array}
  \]
\end{Def}
Die Syntax legt zunächst nur die erlaubten Programme fest. Ihre Bedeutung
definieren wir als Transformation des gesamten Registerzustands.
\begin{Def}[Semantik eines LOOP-Programms]
  Sei $P$ ein LOOP-Programm und $k \ge \max\{ i \mid x_i \text{ kommt in $P$ vor} \}$. Dann definiert $P$ eine Funktion $\llbracket P \rrbracket_k : \N^k \to \N^k$ wie folgt.
  \begin{itemize}
  \item $\llbracket x_i := x_i+1 \rrbracket_k (a_1, \dots, a_i, \dots, a_k) =
    (a_1, \dots, a_i+1, \dots, a_k)$
  \item $\llbracket x_i := x_i\dotminus 1 \rrbracket_k (a_1, \dots, a_i, \dots, a_k) =
    (a_1, \dots, a_i\dotminus 1, \dots, a_k)$
  \item $\llbracket P;Q \rrbracket_k (\overline{a}) =
    \llbracket Q \rrbracket_k (\llbracket P \rrbracket_k (\overline{a}))$
  \item $\llbracket \texttt{LOOP }x_i\texttt{ DO }P\texttt{ END} \rrbracket_k
    (\overline{a}) = \llbracket P \rrbracket_k^{(a_i)} (\overline{a})$
  \end{itemize}


  Ein LOOP-Programm $P$ \emph{berechnet} eine Funktion $f:\N^n\to\N$,
  wenn mit
  \[
    k=\max\bigl(\{n\}\cup\{i\mid x_i\text{ kommt in }P\text{ vor}\}\bigr)
    \qquad\text{und}\qquad
    \operatorname{init}_k(\overline{a})=(a_1,\dots,a_n,0,\dots,0)\in\N^k
  \]
  für jede Eingabe $\overline{a}\in\N^n$ gilt:
  \[
    f(\overline{a})
    =\pi^k_1\bigl(\llbracket P\rrbracket_k
      (\operatorname{init}_k(\overline{a}))\bigr).
  \]
\end{Def}

In Worten: Die Zuweisungen und die sequenzielle Ausführung haben ihre übliche
Bedeutung. Bei der Ausführung von
$\texttt{LOOP }x_i\texttt{ DO }P\texttt{ END}$ wird der Wert $n$ von $x_i$
beim Eintritt in die Schleife einmal gelesen und anschließend $P$ genau
$n$-mal ausgeführt. Änderungen von $x_i$ innerhalb von $P$ beeinflussen die
Anzahl der Schleifendurchläufe nicht.

Die Definition der Semantik von LOOP-Programmen ist
\emph{kompositionell}, weil Semantik einer zusammengesetzten Anweisung
als Funktion der Semantik ihrer Teilanweisungen definiert ist.

\begin{Satz}
  Eine Funktion $f : \N^k \to \N$ ist LOOP-berechenbar genau dann,
  wenn sie primitiv rekursiv ist.
\end{Satz}

Beschränkte Schleifen terminieren stets und können daher nur totale Funktionen
berechnen. Für allgemeine partielle Berechnungen erweitern wir die Sprache um
eine Schleife mit datenabhängiger Abbruchbedingung.
\begin{Def}[Syntax von WHILE-Programmen]
  Ein WHILE-Programm arbeitet mit Registern
  $x_1,x_2,\dots$, deren Werte natürliche Zahlen sind. Die Menge der
  WHILE-Programme ist induktiv durch die folgende Syntax definiert:
  \[
    \begin{array}{rcll}
      P,Q &::=& x_i := x_i+1
      & \text{(Inkrementierung)},\\
          &\mid& x_i := x_i\dotminus 1
      & \text{(Dekrementierung mit Abschneiden bei $0$)},\\
          &\mid& P;Q
      & \text{(sequenzielle Ausführung)},\\
          &\mid& \texttt{LOOP }x_i\texttt{ DO }P\texttt{ END}
      & \text{(beschränkte Schleife)},\\
          &\mid& \texttt{WHILE }x_i\ne0\texttt{ DO }P\texttt{ END}
                 & \text{(unbeschränkte Schleife)}.
    \end{array}
  \]
\end{Def}
Da eine unbeschränkte Schleife nicht terminieren muss, ist ihre Semantik im
Allgemeinen nur eine partielle Funktion auf Registerzuständen.
\begin{Def}[Semantik von WHILE-Programmen]
  Die Semantik wird durch strukturelle Rekursion über WHILE-Programme als
  partielle Funktion
  $\llbracket P\rrbracket_k:\N^k\hookrightarrow\N^k$ definiert. Für
  Inkrementierung und Dekrementierung gelten die Gleichungen aus der
  LOOP-Semantik. Sequenzielle Ausführung und beschränkte Schleifen werden
  durch
  \begin{align*}
    \llbracket P;Q\rrbracket_k
      &=\llbracket Q\rrbracket_k\circ\llbracket P\rrbracket_k,\\
    \llbracket \texttt{LOOP }x_i\texttt{ DO }P\texttt{ END}\rrbracket_k
      (\overline{a})
      &=\llbracket P\rrbracket_k^{(a_i)}(\overline{a})
  \end{align*}
  definiert, wobei Komposition und Iteration nun partiell sein können.

  Für die unbeschränkte Schleife sei
  \[
    H_{P,i}(\overline{a})
    :=\bigl\{n\in\N\bigm|
      \llbracket P\rrbracket_k^{(n)}(\overline{a})\downarrow
      \text{ und }
      \pi_i^k\bigl(\llbracket P\rrbracket_k^{(n)}(\overline{a})\bigr)=0
    \bigr\}.
  \]
  Dann ist
  \[
    \llbracket \texttt{WHILE }x_i\ne0\texttt{ DO }P\texttt{ END}
    \rrbracket_k(\overline{a}) =
    \begin{cases}
      \llbracket P\rrbracket_k^{(m)}(\overline{a}),
        & H_{P,i}(\overline{a})\ne\emptyset
          \text{ und }m=\min H_{P,i}(\overline{a}),\\
      \text{undefiniert}, & H_{P,i}(\overline{a})=\emptyset.
    \end{cases}
  \]
  Ist $m=\min H_{P,i}(\overline{a})$, so ist die $i$-te Komponente von
  $\llbracket P\rrbracket_k^{(\ell)}(\overline{a})$ für alle $\ell<m$ von
  $0$ verschieden und von
  $\llbracket P\rrbracket_k^{(m)}(\overline{a})$ gleich $0$.

  Ein WHILE-Programm $P$ \emph{berechnet} die partielle Funktion
  $f:\N^n\hookrightarrow\N$, falls mit $k$ und
  $\operatorname{init}_k$ wie in der LOOP-Semantik für alle
  $\overline{a}\in\N^n$ die Gleichheit partieller Ausdrücke
  \[
    f(\overline{a})
    =\pi^k_1\bigl(\llbracket P\rrbracket_k
      (\operatorname{init}_k(\overline{a}))\bigr)
  \]
  gilt. Insbesondere ist $f(\overline{a})$ genau dann undefiniert, wenn die
  Ausführung von $P$ auf $\operatorname{init}_k(\overline{a})$ nicht
  terminiert.
\end{Def}
\begin{Satz}
  Eine partielle Funktion $f : \N^k \hookrightarrow \N$ ist
  WHILE-berechenbar genau dann,
  wenn sie $\mu$-rekursiv ist.
\end{Satz}

Eine genauere Normalform zeigt, dass für jede $\mu$-rekursive Funktion eine
einzige unbeschränkte Suche genügt; davor und danach stehen nur primitiv
rekursive Berechnungen.
\begin{Satz}[name={Kleenes Normalformtheorem}]
	Für jede partielle Funktion $F:\N^k\hookrightarrow\N$ mit
	$F\in\ac{muREC}$ gibt es primitiv rekursive Funktionen
	$p:\N\to\N$ und $q:\N^{k+1}\to\N$, sodass
	\[
		F(\overline{x})=p\bigl((\mu q)(\overline{x})\bigr)
	\]
	als Gleichheit partieller Funktionen gilt.
\end{Satz}
Zusammen mit der Äquivalenz von $\mu$-Rekursion und Turing-Berechenbarkeit folgt
insbesondere: Jede berechenbare Funktion über $\N$ kann mit einem Programm aus
der \acf{WHILE} berechnet werden, die maximal eine WHILE-Anweisung
verwendet.

Zum Abschluss kehren wir zu den totalen Funktionen zurück. Die Ackermannfunktion
ist berechenbar, wächst jedoch zu schnell, um primitiv rekursiv zu sein.
\begin{Def}[name={[Ackermannfunktion]}]
	Die \emph{Ackermannfunktion}
	\begin{align*}
		&A:\N^2\->\N\\
		&A(0,y)=y+1\\
		x>0:\ &A(x,y)=A(x-1,1)\\
		y>0:\ &A(x,y)=A(x-1,A(x.y-1)) \tag*{\qedhere}
	\end{align*}
\end{Def}
Um diese Behauptung zu präzisieren, vergleichen wir die Wachstumsraten
primitiv rekursiver Funktionen mit der Ackermannfunktion.
\begin{Satz}
	Sei $\A$ die Menge der Funktionen, die durch $A$ majorisiert werden.
	\[ \text{Es gilt: }\ac{PREC}\subseteq \A \]\vspace{-2em}
\end{Satz}
\begin{proof}
	$A=\{f:\N^k\->\N \mid \exists n \forall\overline{x}:f(\overline{x})< A(m,\max\overline{x})\}$.\\
	Induktion über $\ac{PREC}$:
	\begin{enumerate}
	\item $\mathcal{O}(\overline{x})=0 < A(0,\max\overline{x})=\max \overline{x}+1,\quad \mathcal{O}\in\A$
	\item $S(x)=\A+1 < x+2 = A(\underline{1},x)$, also $S\in \mathcal A$
	\item $\pi^k_j(x_1,\dots,x_k) = x_j\leq \max\overline{x} < \max \overline{x} +1 = A(\underline{0},\max \overline{x})$, also $\pi^k_j\in\A$
	\item\ \vspace{-2em}
		\begin{align*}
			&\left.\begin{array}{rr<{{}}@{}l}
				\text{Sei } & g_1,\dots,g_m : &\N^k\;\->\N\\
				& h : &\N^m\->\N
			\end{array}\right\rbrace \in\A\\
			&\begin{array}{rr@{}>{{}}l@{\qquad}r@{\;}l}
				\text{also } &g_i(\overline{x}) &< A(r_i,\max\overline{x}) & r_i &\exists \text{ nach I.V.}\\
				&h(\overline{x}) &< A(s,\max\overline{y}) &s &\exists \text{ nach I.V.}
			\end{array}
		\end{align*}
		\newpage
		Betrachte $f= h\circ [g_i,\dots,g_m]$\\
		Wähle $j$ sodass $r_j=\max r_i$\\
		$\curvearrowright\ g_i(x) < A(r_j,\max\overline{x})$
		\begin{align*}
			\text{Dann }f(\overline{x}) &= h(g_1(\overline{x}),\dots,g_m(\overline{x})\\
			&< A(s,\max(g_1(\overline{x}),\dots,g_m(\overline{x}))\\
			&< A(s,\max(A(r_1,\max\overline{x}), \dots, A(r_m,\max\overline{x}))\\
			&= A(s,A(r_j,\max\overline{x}))\\
			&\overset!< A(\underline{s+r_j+2},\max\overline{x})\\
			\text{Also }f\in\A
		\end{align*}
	\item Sei $g:\N^k\->\N,\ h:\N^[k+2]\->\N\qquad \in\A$
		\begin{align*}
			\curvearrowright \exists r,s :{}& g(\overline{x}< A(r,\max\overline{x}))\\
			& h(\overline{y}< A(s,\max\overline{y}))
		\end{align*}
		Sei $f= PR(g,h)$. Zeige $f\in\A$\\
		Zunächst $\forall n\in\N: f(n,\overline{x}) < A(q,n+\max\overline x)$, wobei $q$ nicht von $n$ oder $\overline{x}$ abhängt.\\
		Wähle $q=\max(r,s)+1$.
		
		Induktion über $n$:
		\begin{description}
		\item[$n=0:$]\qquad $f(0,\overline{x}) = g(\overline{x}) < A(r,\max\overline{x}) < A(g,\max x)$
		\item[$n\->n+1:$]\ \vspace{-2em}
			\begin{align*}
				f(n+1,\overline{x}) &= h(f(n,\overline{x}),n,\overline{x}) < A(s,z) &&,\ z=\max(f(n,\overline{x}),n,\overline{x})\\
				\text{nach I.V.\,:}\quad f(n,\overline{x}) &< A(g,n+\max\overline{x})\\
				\max(n,\overline{x}) \leq n+\max\overline{x} &< A(q,n+\max\overline{x})\\
				\curvearrowright z &< A(q,n+\max\overline{x})\\
				f(n+1,\overline{x}) &< A(s,z)\\
				&< A(s,A(q,n+\max\overline{x}))\\
				&\leq A(q-1,A(q,n+\max\overline{x}))\\
				&= A(q,A(q,n+1+\max\overline{x})) \tag*{\qedsymbol\ Ind. $n$}
			\end{align*}
		\end{description}
		\begin{align*}
			\text{Nun setze }w &=\max(n,\overline{x})\\
			f(n,\overline{x}) < A(q,n+\max\overline{x})\\
			&\leq A(q,2w)\\
			&\overset!< A(\underline{q+2},w)\\
			\text{Also }f\in\A \tag*{\qedhere}
		\end{align*}
	\end{enumerate}
\end{proof}
Der Majorisierungssatz liefert unmittelbar die gewünschte echte Trennung.
\begin{Korollar}
	Die Ackermannfunktion $A$ ist nicht primitiv rekursiv.
\end{Korollar}
\begin{proof}
  $A\notin\A$
\end{proof}
